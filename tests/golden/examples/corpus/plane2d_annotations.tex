\section{Plane2D Annotations Demo -- Convex Hull Visualization}

This demo showcases text-only annotations on Plane2D points (the CRITICAL fix),
along with arrow annotations for comparison. We visualize a set of points and
identify candidates for a convex hull.

\begin{diagram}[id="plane2d-annot-demo", label="Convex hull point annotations"]
\shape{p}{Plane2D}{xrange=[-5,5], yrange=[-5,5], grid=true, axes=true, show_coords=true}

% --- Add 8 points at interesting positions ---
\apply{p}{add_point=(0,0)}
\apply{p}{add_point=(3,4)}
\apply{p}{add_point=(-3,2)}
\apply{p}{add_point=(4,-2)}
\apply{p}{add_point=(-2,-4)}
\apply{p}{add_point=(1,3)}
\apply{p}{add_point=(-4,-1)}
\apply{p}{add_point=(2,-3)}

% --- Recolor points by role ---
\recolor{p.point[0]}{state=current}
\recolor{p.point[1]}{state=good}
\recolor{p.point[2]}{state=good}
\recolor{p.point[3]}{state=good}
\recolor{p.point[4]}{state=good}
\recolor{p.point[5]}{state=dim}
\recolor{p.point[6]}{state=good}
\recolor{p.point[7]}{state=error}

% --- Text-only annotations (NO arrow_from) on each point ---
\annotate{p.point[0]}{label="Origin", position=above, color=info}
\annotate{p.point[1]}{label="Hull vertex", position=above, color=good}
\annotate{p.point[2]}{label="Hull vertex", position=left, color=good}
\annotate{p.point[3]}{label="Hull vertex", position=right, color=good}
\annotate{p.point[4]}{label="Hull vertex", position=below, color=good}
\annotate{p.point[5]}{label="Interior pt", position=right, color=warn}
\annotate{p.point[6]}{label="Hull vertex", position=left, color=good}
\annotate{p.point[7]}{label="Outlier", position=below, color=error}

% --- Arrow annotations for comparison (2 examples) ---
\annotate{p.point[5]}{label="inside hull", arrow_from="p.point[0]", color=warn}
\annotate{p.point[7]}{label="far from centroid", arrow_from="p.point[0]", color=error}

\end{diagram}

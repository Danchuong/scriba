\subsection{Code Block}

\begin{lstlisting}[language=Python]
def prefix_sum(arr):
    n = len(arr)
    ps = [0] * (n + 1)
    for i in range(n):
        ps[i + 1] = ps[i] + arr[i]
    return ps
\end{lstlisting}

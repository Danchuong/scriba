\step
\narrate{Khoi tao Disjoint-Set Union (DSU) tren 7 phan tu A..G. Moi phan tu la goc cua chinh no: $parent[x] = x$, $rank[x] = 0$. Chua co mui ten nao -- moi node deu doc lap. Ta se dung annotation arrows giua cac o cua mang $parent[]$ de the hien quan he cha-con.}

\step
\recolor{parent.cell[0]}{state=current}

\recolor{parent.cell[1]}{state=good}
\annotate{parent.cell[0]}{label="B→A", arrow_from="parent.cell[1]", color=good}
\narrate{$union(A, B)$: goc cua A la A, goc cua B la B. Rank bang nhau nen gan B lam con cua A: $parent[1] = A$. Ve mui ten tu cell[1] den cell[0] (B tro ve A). Thanh phan: $\{A, B\}$ voi goc A.}

\step
\recolor{parent.cell[0]}{state=done}

\narrate{$union(C, D)$: goc C va goc D khac nhau. Gan D lam con cua C: $parent[3] = C$. Mui ten cell[3] $\to$ cell[2]. Thanh phan: $\{A,B\}, \{C,D\}, \{E\}, \{F\}, \{G\}$.}

\step

\narrate{$union(E, F)$: gan F lam con cua E: $parent[5] = E$. Mui ten cell[5] $\to$ cell[4]. Thanh phan: $\{A,B\}, \{C,D\}, \{E,F\}, \{G\}$.}

\step
\recolor{parent.cell[4]}{state=done}
\recolor{parent.cell[5]}{state=done}
\recolor{parent.cell[0]}{state=current}
\recolor{parent.cell[2]}{state=current}
\apply{parent.cell[2]}{value="A"}
\recolor{parent.cell[2]}{state=good}
\apply{rank.cell[0]}{value=1}
\recolor{rank.cell[0]}{state=good}
\annotate{parent.cell[0]}{label="C→A", arrow_from="parent.cell[2]", color=good}
\narrate{$union(A, C)$: $find(A) = A$, $find(C) = C$. Goc khac nhau. Rank bang nhau nen gan C lam con cua A: $parent[2] = A$, $rank[0] = 1$. Mui ten cell[2] $\to$ cell[0]. Bay gio D la con cua C, C la con cua A. Tim goc cua D can di $D \to C \to A$ (2 buoc). Day la chuoi se duoc nen boi path compression.}

\step
\recolor{parent.cell[0]}{state=done}
\recolor{parent.cell[2]}{state=done}
\recolor{rank.cell[0]}{state=done}
\recolor{parent.cell[3]}{state=current}
\narrate{$find(D)$ voi path compression -- Buoc 1: bat dau tai D (cell[3]). $parent[3] = C$. Di theo mui ten cell[3] $\to$ cell[2] (D tro ve C). D chua phai goc, can tiep tuc.}

\step
\recolor{parent.cell[3]}{state=dim}
\recolor{parent.cell[2]}{state=current}
\narrate{$find(D)$ -- Buoc 2: tu C (cell[2]). $parent[2] = A$. Di theo mui ten cell[2] $\to$ cell[0] (C tro ve A). Tiep tuc tim goc.}

\step
\recolor{parent.cell[2]}{state=dim}
\recolor{parent.cell[0]}{state=good}
\narrate{$find(D)$ -- Buoc 3: den A (cell[0]). $parent[0] = A$, day la goc! $find(D)$ tra ve A. Bay gio thuc hien path compression: doi parent cua D tu C thanh A, xoa mui ten cu va ve mui ten moi truc tiep den goc.}

\step
\apply{parent.cell[3]}{value="A"}
\recolor{parent.cell[3]}{state=good}
\recolor{parent.cell[0]}{state=good}
\recolor{parent.cell[2]}{state=done}
\reannotate{parent.cell[0]}{label="D→A (compressed)", arrow_from="parent.cell[3]", color=path}
\narrate{Path compression hoan tat: $parent[3] = A$. Mui ten cu cell[3] $\to$ cell[2] (D $\to$ C) da duoc thay the boi mui ten moi cell[3] $\to$ cell[0] (D $\to$ A, mau path). Lan goi $find(D)$ tiep theo se la $O(1)$ vi D tro truc tiep den goc A.}

\step

\narrate{$union(A, E)$: $find(A) = A$, $find(E) = E$. Goc khac nhau. $rank[0] = 1 > rank[4] = 0$, nen gan E lam con cua A: $parent[4] = A$. Mui ten cell[4] $\to$ cell[0]. Bay gio A la goc cua tat ca tru G. F van tro ve E ($parent[5] = E$), chua duoc nen.}

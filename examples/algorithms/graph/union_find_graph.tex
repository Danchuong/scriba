\begin{animation}[id="union-find-graph", label="Union-Find (DSU) -- Graph primitive voi vi tri co dinh"]
\shape{G}{Graph}{nodes=["A","B","C","D","E","F","G"], edges=[], directed=true, layout="stable", layout_seed=42}
\shape{parent}{Array}{size=7, data=["A","B","C","D","E","F","G"], labels="0..6", label="$parent[]$"}

\step
\narrate{Disjoint-Set Union (DSU) tren 7 phan tu A..G. Ban dau moi phan tu la goc cua chinh no: $parent[x] = x$. Do thi co huong khong co canh nao -- 7 thanh phan doc lap. Cac node duoc dat co dinh tren mot hang (row 0, y=0). Chi co canh thay doi, node khong di chuyen.}

\step
\recolor{G.node[A]}{state=current}
\recolor{G.node[B]}{state=current}
\apply{G}{add_edge={from="B", to="A"}}
\apply{parent.cell[1]}{value="A"}
\recolor{parent.cell[1]}{state=good}
\narrate{$union(A, B)$: goc cua A la A, goc cua B la B. Lien ket B duoi A: them canh $B \to A$, dat $parent[1] = A$. Thanh phan: $\{A, B\}$ voi goc A. So canh: $0 \to 1$.}

\step
\recolor{G.node[A]}{state=done}
\recolor{G.node[B]}{state=done}
\recolor{G.node[C]}{state=current}
\recolor{G.node[D]}{state=current}
\apply{G}{add_edge={from="D", to="C"}}
\apply{parent.cell[3]}{value="C"}
\recolor{parent.cell[3]}{state=good}
\narrate{$union(C, D)$: lien ket D duoi C. Them canh $D \to C$, $parent[3] = C$. Thanh phan: $\{A,B\}, \{C,D\}, \{E\}, \{F\}, \{G\}$. So canh: $1 \to 2$.}

\step
\recolor{G.node[C]}{state=done}
\recolor{G.node[D]}{state=done}
\recolor{G.node[E]}{state=current}
\recolor{G.node[F]}{state=current}
\apply{G}{add_edge={from="F", to="E"}}
\apply{parent.cell[5]}{value="E"}
\recolor{parent.cell[5]}{state=good}
\narrate{$union(E, F)$: lien ket F duoi E. Them canh $F \to E$, $parent[5] = E$. Thanh phan: $\{A,B\}, \{C,D\}, \{E,F\}, \{G\}$. So canh: $2 \to 3$.}

\step
\recolor{G.node[E]}{state=done}
\recolor{G.node[F]}{state=done}
\recolor{G.node[A]}{state=current}
\recolor{G.node[C]}{state=current}
\apply{G}{add_edge={from="C", to="A"}}
\apply{parent.cell[2]}{value="A"}
\recolor{parent.cell[2]}{state=good}
\narrate{$union(A, C)$: $find(A) = A$, $find(C) = C$. Lien ket goc C duoi goc A: them canh $C \to A$, $parent[2] = A$. Cay hien tai: $A \leftarrow B$, $A \leftarrow C \leftarrow D$. Tim goc cua D can 2 buoc: $D \to C \to A$. Day la chuoi se bi nen boi path compression. So canh: $3 \to 4$.}

\step
\recolor{G.node[A]}{state=done}
\recolor{G.node[C]}{state=dim}
\recolor{G.node[D]}{state=current}
\narrate{$find(D)$ voi path compression -- buoc 1: bat dau tai D. $parent[D] = C$, di theo canh $D \to C$. Ghi nho D la node can duoc gan lai cha truc tiep.}

\step
\recolor{G.node[D]}{state=path}
\recolor{G.node[C]}{state=current}
\narrate{$find(D)$ buoc 2: tu C, $parent[C] = A$, di theo canh $C \to A$. C da tro truc tiep den goc nen khong can nen.}

\step
\recolor{G.node[C]}{state=path}
\recolor{G.node[A]}{state=good}
\narrate{$find(D)$ buoc 3: den A, $parent[A] = A$ -- day la goc. $find(D)$ tra ve A. Bay gio thuc hien path compression: xoa canh $D \to C$, them canh $D \to A$ de D tro thang den goc.}

\step
\apply{G}{remove_edge={from="D", to="C"}}
\apply{G}{add_edge={from="D", to="A"}}
\apply{parent.cell[3]}{value="A"}
\recolor{parent.cell[3]}{state=good}
\recolor{G.node[D]}{state=done}
\recolor{G.node[C]}{state=done}
\recolor{G.node[A]}{state=done}
\narrate{Path compression hoan tat: $remove\_edge(D \to C)$ roi $add\_edge(D \to A)$. $parent[3] = A$. Cac lan goi $find(D)$ tiep theo se la $O(1)$. So canh: $4 \to 3 \to 4$ (xoa/them cap). Node C van tro den A, chi co D duoc gan lai.}

\step
\recolor{G.node[A]}{state=current}
\recolor{G.node[E]}{state=current}
\apply{G}{add_edge={from="E", to="A"}}
\apply{parent.cell[4]}{value="A"}
\recolor{parent.cell[4]}{state=good}
\narrate{$union(A, E)$: $find(A) = A$, $find(E) = E$. Lien ket goc E duoi goc A: them canh $E \to A$, $parent[4] = A$. Tat ca 7 phan tu gio thuoc cung mot thanh phan voi goc A. Duong di cua F: $F \to E \to A$ (2 buoc), G van doc lap cua cay E. So canh: $4 \to 5$.}

\step
\recolor{G.node[A]}{state=good}
\recolor{G.node[B]}{state=done}
\recolor{G.node[C]}{state=done}
\recolor{G.node[D]}{state=done}
\recolor{G.node[E]}{state=done}
\recolor{G.node[F]}{state=done}
\recolor{G.node[G]}{state=done}
\recolor{parent.all}{state=done}
\recolor{parent.cell[0]}{state=good}
\narrate{Trang thai cuoi cung: tat ca 7 phan tu nam trong mot thanh phan duy nhat voi goc A. $parent[] = [A, A, A, A, A, E, G]$. Node G va F chua duoc goi $find()$ nen van giu nguyen cha cu ($G$ la goc cua chinh no, $F \to E$). Day chinh la ban chat cua DSU: chi nen duong di khi $find$ duoc goi, do do chi phi phan bo la $O(\alpha(n))$ moi thao tac. Toan bo qua trinh chi thay doi canh -- vi tri node co dinh, the hien dung cau truc logic cua Graph primitive.}

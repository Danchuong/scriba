\begin{animation}[id="bfs", label="BFS Traversal"]
\shape{G}{Graph}{nodes=[A,B,C,D,E], edges=[(A,B),(A,C),(B,D),(C,D),(C,E)]}

\step
\recolor{G.node[A]}{state=current}
\narrate{Start BFS from node A. Add A to the queue.}

\step
\recolor{G.node[A]}{state=done}
\recolor{G.node[B]}{state=current}
\recolor{G.node[C]}{state=current}
\recolor{G.edge[(A,B)]}{state=done}
\recolor{G.edge[(A,C)]}{state=done}
\narrate{Visit A. Enqueue neighbors B and C.}

\step
\recolor{G.node[B]}{state=done}
\recolor{G.node[D]}{state=current}
\recolor{G.edge[(B,D)]}{state=done}
\narrate{Visit B. Enqueue neighbor D (C already seen).}

\step
\recolor{G.node[C]}{state=done}
\recolor{G.node[D]}{state=done}
\recolor{G.node[E]}{state=current}
\recolor{G.edge[(C,D)]}{state=done}
\recolor{G.edge[(C,E)]}{state=done}
\narrate{Visit C then D. Enqueue E. Visit E. BFS complete.}
\end{animation}

\begin{animation}[id="selsort", label="Selection Sort"]
\shape{a}{Array}{size=6, data=[29,10,14,37,13,8], labels="0..5"}

\step
\highlight{a.cell[5]}
\recolor{a.range[0:5]}{state=current}
\narrate{Pass 1: scan entire array for minimum. Min = 8 at index 5.}

\step
\apply{a.cell[0]}{value=8}
\apply{a.cell[5]}{value=29}
\recolor{a.cell[0]}{state=done}
\recolor{a.range[1:5]}{state=idle}
\narrate{Swap a[0] and a[5]. Array: [8, 10, 14, 37, 13, 29]. a[0] is sorted.}

\step
\recolor{a.range[1:5]}{state=current}
\highlight{a.cell[1]}
\narrate{Pass 2: scan a[1..5] for minimum. Min = 10 at index 1. Already in place.}

\step
\recolor{a.cell[1]}{state=done}
\recolor{a.range[2:5]}{state=current}
\highlight{a.cell[4]}
\narrate{Pass 3: scan a[2..5] for minimum. Min = 13 at index 4.}

\step
\apply{a.cell[2]}{value=13}
\apply{a.cell[4]}{value=14}
\recolor{a.cell[2]}{state=done}
\recolor{a.range[3:5]}{state=current}
\highlight{a.cell[5]}
\narrate{Swap a[2] and a[4]. Array: [8, 10, 13, 37, 14, 29]. Scan a[3..5], min = 14 at index 4.}

\step
\apply{a.cell[3]}{value=14}
\apply{a.cell[4]}{value=37}
\recolor{a.cell[3]}{state=done}
\recolor{a.range[4:5]}{state=current}
\highlight{a.cell[4]}
\narrate{Swap a[3] and a[4]. Array: [8, 10, 13, 14, 37, 29]. Scan a[4..5], min = 29 at index 5.}

\step
\apply{a.cell[4]}{value=29}
\apply{a.cell[5]}{value=37}
\recolor{a.cell[4]}{state=done}
\recolor{a.cell[5]}{state=done}
\narrate{Swap a[4] and a[5]. Final sorted array: [8, 10, 13, 14, 29, 37].}
\end{animation}

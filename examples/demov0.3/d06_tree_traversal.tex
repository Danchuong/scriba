\begin{animation}[id="tree-inorder", label="In-Order Binary Tree Traversal"]
\shape{T}{Tree}{root=4, nodes=[4,2,6,1,3,5,7], edges=[(4,2),(4,6),(2,1),(2,3),(6,5),(6,7)]}

\step
\narrate{Binary tree with 7 nodes. In-order traversal visits left subtree, root, then right subtree.}

\step
\recolor{T.node[4]}{state=dim}
\recolor{T.node[2]}{state=dim}
\recolor{T.node[1]}{state=current}
\narrate{Go left from 4, left from 2, reach leaf 1. Visit node 1.}

\step
\recolor{T.node[1]}{state=done}
\recolor{T.node[2]}{state=current}
\narrate{Backtrack. Visit node 2. Output so far: 1, 2.}

\step
\recolor{T.node[2]}{state=done}
\recolor{T.node[3]}{state=current}
\narrate{Visit right child of 2. Visit node 3. Output: 1, 2, 3.}

\step
\recolor{T.node[3]}{state=done}
\recolor{T.node[4]}{state=current}
\narrate{Left subtree done. Visit root node 4. Output: 1, 2, 3, 4.}

\step
\recolor{T.node[4]}{state=done}
\recolor{T.node[6]}{state=dim}
\recolor{T.node[5]}{state=current}
\narrate{Go right to 6, then left to 5. Visit node 5. Output: 1, 2, 3, 4, 5.}

\step
\recolor{T.node[5]}{state=done}
\recolor{T.node[6]}{state=current}
\narrate{Visit node 6. Output: 1, 2, 3, 4, 5, 6.}

\step
\recolor{T.node[6]}{state=done}
\recolor{T.node[7]}{state=current}
\narrate{Visit right child 7. Output: 1, 2, 3, 4, 5, 6, 7. Traversal complete.}
\end{animation}

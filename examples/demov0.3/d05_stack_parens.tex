\begin{animation}[id="stack-parens", label="Parenthesis Matching with Stack"]
\shape{input}{Array}{size=8, data=["(","(",")",")","(","(",")",")"]}
\shape{s}{Stack}{items=[]}

\step
\highlight{input.cell[0]}
\narrate{Input: ( ( ) ) ( ( ) ). Push ( on open, pop on close match.}

\step
\recolor{input.cell[0]}{state=current}
\apply{s}{push="("}
\narrate{Char 1: ( is open. PUSH. Stack: [(.]}

\step
\recolor{input.cell[0]}{state=done}
\recolor{input.cell[1]}{state=current}
\apply{s}{push="("}
\narrate{Char 2: ( is open. PUSH. Stack: [(, (.]}

\step
\recolor{input.cell[1]}{state=done}
\recolor{input.cell[2]}{state=current}
\apply{s}{pop=1}
\narrate{Char 3: ) closes. POP top (. Match! Stack: [(.]}

\step
\recolor{input.cell[2]}{state=done}
\recolor{input.cell[3]}{state=current}
\apply{s}{pop=1}
\narrate{Char 4: ) closes. POP top (. Match! Stack: empty.}

\step
\recolor{input.cell[3]}{state=done}
\recolor{input.cell[4]}{state=current}
\apply{s}{push="("}
\narrate{Char 5: ( is open. PUSH. Stack: [(.]}

\step
\recolor{input.cell[4]}{state=done}
\recolor{input.cell[5]}{state=current}
\apply{s}{push="("}
\narrate{Char 6: ( is open. PUSH. Stack: [(, (.]}

\step
\recolor{input.cell[5]}{state=done}
\recolor{input.cell[6]}{state=current}
\apply{s}{pop=1}
\narrate{Char 7: ) closes. POP top (. Match! Stack: [(.]}

\step
\recolor{input.cell[6]}{state=done}
\recolor{input.cell[7]}{state=current}
\apply{s}{pop=1}
\narrate{Char 8: ) closes. POP top (. Match! Stack: empty.}

\step
\recolor{input.cell[7]}{state=done}
\recolor{input.range[0:7]}{state=good}
\narrate{All 8 chars processed. Stack empty = BALANCED parentheses.}
\end{animation}

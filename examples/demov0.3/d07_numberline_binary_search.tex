\step
\recolor{nl.range[12:15]}{state=dim}

\narrate{Mid = 11. Target 11 equals mid. Found it!}

\step
\recolor{nl.range[0:10]}{state=dim}
\recolor{nl.range[12:15]}{state=dim}
\recolor{nl.range[11:11]}{state=good}
\narrate{Binary search complete. Target 11 found in 2 comparisons. O(log n) efficiency.}

% Regression fixture: C1 — Cumulative budget never wired.
%
% This file verifies that a document with many \compute blocks each doing
% sub-budget work is still capped by the cumulative wall-clock budget.
%
% Each \compute block runs a tight Python loop to burn ~300 ms of wall-clock
% time (using only bytecode operations so SIGALRM fires between steps).
% Twenty such blocks would accumulate ~6 s total, which exceeds the 5 s
% cumulative budget (_CUMULATIVE_BUDGET_SECONDS = 5.0).
%
% Expected behaviour (with fix):
%   The render raises WorkerError with code E1152 containing "cumulative"
%   before the 18th block completes.
%
% Expected behaviour (without fix):
%   All 20 blocks complete unchecked, accumulating ~6 s with zero error.

\step

for i in range(250000):
    x = x + 1
result = x}
\narrate{Block 1 done.}

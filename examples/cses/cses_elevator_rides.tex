\begin{animation}[id="elevator-rides", label="Elevator Rides -- Bitmask DP (n=4, x=8)"]
\shape{w}{Array}{size=4, data=[3,5,2,7], labels="0..3", label="weights"}
\shape{dp}{DPTable}{n=16, labels="0..15", label="dp[mask] = (rides, remaining)"}

\step
\narrate{Problem: 4 people with weights [3, 5, 2, 7], elevator capacity x = 8. Find the minimum number of rides. We use bitmask DP where each of the 16 masks represents a subset of people already transported. dp[mask] stores (min rides, max remaining capacity in last ride).}

\step
\apply{dp.cell[0]}{value="1,8"}
\recolor{dp.cell[0]}{state=good}
\narrate{Base case: dp[0] = (1, 8). No one transported yet. We have 1 ride open with full capacity 8 remaining. Mask 0 = binary 0000 means no person selected.}

\step
\highlight{w.cell[0]}
\recolor{dp.cell[1]}{state=current}
\apply{dp.cell[1]}{value="1,5"}
\narrate{Try adding person 0 (weight 3) to mask 0000. Weight 3 <= 8 remaining, so they fit in the current ride. dp[0001] = (1, 8-3) = (1, 5). One ride used, 5 capacity left.}

\step
\recolor{dp.cell[1]}{state=done}
\highlight{w.cell[1]}
\recolor{dp.cell[2]}{state=current}
\apply{dp.cell[2]}{value="1,3"}
\narrate{Add person 1 (weight 5) to mask 0000. Weight 5 <= 8, fits. dp[0010] = (1, 8-5) = (1, 3). One ride, 3 remaining.}

\step
\recolor{dp.cell[2]}{state=done}
\highlight{w.cell[2]}
\recolor{dp.cell[4]}{state=current}
\apply{dp.cell[4]}{value="1,6"}
\narrate{Add person 2 (weight 2) to mask 0000. Weight 2 <= 8, fits. dp[0100] = (1, 8-2) = (1, 6).}

\step
\recolor{dp.cell[4]}{state=done}
\highlight{w.cell[3]}
\recolor{dp.cell[8]}{state=current}
\apply{dp.cell[8]}{value="1,1"}
\narrate{Add person 3 (weight 7) to mask 0000. Weight 7 <= 8, fits. dp[1000] = (1, 8-7) = (1, 1). All four single-person states computed from the empty set.}

\step
\recolor{dp.cell[8]}{state=done}
\recolor{dp.cell[1]}{state=current}
\highlight{w.cell[1]}
\recolor{dp.cell[3]}{state=current}
\apply{dp.cell[3]}{value="1,0"}
\narrate{Now process mask 0001 = (1,5). Add person 1 (weight 5). 5 <= 5 remaining, fits! dp[0011] = (1, 5-5) = (1, 0). Persons 0 and 1 share one ride at exactly capacity 3+5=8.}

\step
\recolor{dp.cell[3]}{state=done}
\recolor{dp.cell[1]}{state=current}
\highlight{w.cell[2]}
\recolor{dp.cell[5]}{state=current}
\apply{dp.cell[5]}{value="1,3"}
\narrate{From mask 0001 = (1,5), add person 2 (weight 2). 2 <= 5, fits. dp[0101] = (1, 5-2) = (1, 3). Persons 0 and 2 share a ride (weight 3+2=5).}

\step
\recolor{dp.cell[5]}{state=done}
\recolor{dp.cell[1]}{state=current}
\highlight{w.cell[3]}
\recolor{dp.cell[9]}{state=current}
\apply{dp.cell[9]}{value="2,1"}
\narrate{From mask 0001 = (1,5), add person 3 (weight 7). 7 > 5 remaining -- does not fit! Start a new ride. dp[1001] = (1+1, 8-7) = (2, 1). Now we need 2 rides.}

\step
\recolor{dp.cell[9]}{state=done}
\recolor{dp.cell[1]}{state=done}
\recolor{dp.cell[2]}{state=current}
\highlight{w.cell[2]}
\recolor{dp.cell[6]}{state=current}
\apply{dp.cell[6]}{value="1,1"}
\narrate{From mask 0010 = (1,3), add person 2 (weight 2). 2 <= 3, fits. dp[0110] = (1, 3-2) = (1, 1). Persons 1 and 2 share a ride (weight 5+2=7).}

\step
\recolor{dp.cell[6]}{state=done}
\recolor{dp.cell[2]}{state=done}
\highlight{w.cell[3]}
\recolor{dp.cell[10]}{state=current}
\apply{dp.cell[10]}{value="2,1"}
\narrate{From mask 0010 = (1,3), add person 3 (weight 7). 7 > 3, new ride. dp[1010] = (2, 8-7) = (2, 1). Persons 1 and 3 each need their own ride.}

\step
\recolor{dp.cell[10]}{state=done}
\recolor{dp.cell[4]}{state=current}
\highlight{w.cell[3]}
\recolor{dp.cell[12]}{state=current}
\apply{dp.cell[12]}{value="2,1"}
\narrate{From mask 0100 = (1,6), add person 3 (weight 7). 7 > 6, new ride. dp[1100] = (2, 1). Note: also reachable from mask 1000+person 2 giving (2, 6) which is better -- we will update next.}

\step
\recolor{dp.cell[12]}{state=current}
\apply{dp.cell[12]}{value="2,6"}
\narrate{Update! From mask 1000 = (1,1), add person 2 (weight 2). 2 > 1, new ride. Candidate = (2, 8-2) = (2, 6). Since 6 > 1 remaining and same rides, dp[1100] updates to (2, 6). More remaining capacity means more room for future people.}

\step
\recolor{dp.cell[12]}{state=done}
\recolor{dp.cell[4]}{state=done}
\recolor{dp.cell[8]}{state=done}
\recolor{dp.cell[3]}{state=current}
\highlight{w.cell[2]}
\recolor{dp.cell[7]}{state=current}
\apply{dp.cell[7]}{value="2,6"}
\narrate{From mask 0011 = (1,0), add person 2 (weight 2). 2 > 0, new ride. dp[0111] = (2, 8-2) = (2, 6). Persons 0,1 filled ride 1 completely; person 2 starts ride 2 with 6 remaining.}

\step
\recolor{dp.cell[7]}{state=done}
\recolor{dp.cell[3]}{state=done}
\highlight{w.cell[3]}
\recolor{dp.cell[11]}{state=current}
\apply{dp.cell[11]}{value="2,0"}
\narrate{Best path to mask 0011 = (1,0), add person 3 (weight 7). 7 > 0, new ride. dp[1011] = (2, 8-7) = (2, 1). But also from mask 0101=(1,3)+person 3: 7>3, gives (2,1). And from mask 1001=(2,1)+person 1: 5>1, gives (3,3). Best so far: (2,1). Actually from mask 0110=(1,1)+person 3: 7>1, gives (2,1). All paths agree on 2 rides.}

\step
\recolor{dp.cell[11]}{state=done}
\recolor{dp.cell[6]}{state=current}
\highlight{w.cell[3]}
\recolor{dp.cell[14]}{state=current}
\apply{dp.cell[14]}{value="2,0"}
\narrate{From mask 0110 = (1,1), add person 3 (weight 7). 7 > 1, new ride. dp[1110] = (2, 8-7) = (2, 1). But from mask 1100=(2,6)+person 1: 5<=6, fits! Candidate (2,1). And from mask 1010=(2,1)+person 2: 2>1, gives (3,6). Best: (2, 1). Then from mask 0110=(1,1)+person 3: (2,1). All yield 2 rides.}

\step
\recolor{dp.cell[14]}{state=done}
\recolor{dp.cell[6]}{state=done}
\recolor{dp.cell[13]}{state=current}
\apply{dp.cell[13]}{value="2,3"}
\narrate{dp[1101] = mask with persons {0,2,3}. Best path: from mask 0101=(1,3)+person 3: 7>3, gives (2,1). From mask 1100=(2,6)+person 0: 3<=6, gives (2,3). The second is better -- same 2 rides but 3 remaining beats 1. dp[1101] = (2, 3).}

\step
\recolor{dp.cell[13]}{state=done}
\recolor{dp.cell[15]}{state=current}
\narrate{Finally, dp[1111] -- all 4 people. We check transitions from all masks missing exactly one person. From dp[0111]=(2,6)+person 3 (w=7): 7>6, new ride, candidate (3,1). From dp[1011]=(2,1)+person 2 (w=2): 2>1, new ride, candidate (3,6). From dp[1101]=(2,3)+person 1 (w=5): 5>3, new ride, candidate (3,3). From dp[1110]=(2,1)+person 0 (w=3): 3>1, new ride, candidate (3,3).}

\step
\apply{dp.cell[15]}{value="3,6"}
\narrate{All four transitions give 3 rides. The best remaining is 6 (from dp[1011]+person 2). dp[1111] = (3, 6). But the answer is just the number of rides: 3.}

\step
\recolor{dp.cell[15]}{state=good}
\recolor{dp.range[0:14]}{state=done}
\narrate{Answer: dp[1111].rides = 3. The minimum number of elevator rides for weights [3, 5, 2, 7] with capacity 8 is 3. One optimal grouping: ride 1 = {person 1, person 0} (5+3=8), ride 2 = {person 2} (2), ride 3 = {person 3} (7). The bitmask DP explored all 2^4 = 16 subsets in O(2^n * n) time.}

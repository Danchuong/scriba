\begin{animation}[id="repetitions", label="Longest Consecutive Repetition in DNA"]

\compute{
  s = ["A", "T", "T", "C", "G", "G", "G", "A"]
  n = len(s)
}

\shape{dna}{Array}{size=8, data=["A", "T", "T", "C", "G", "G", "G", "A"], labels="0..7", label="DNA string"}
\shape{vars}{Array}{size=2, data=[1, 1], labels="0..1", label="current | max"}

\step
\recolor{dna.cell[0]}{state=current}
\narrate{Bắt đầu duyệt. Ký tự đầu tiên: A. Đoạn lặp hiện tại = 1, max = 1.}

\step
\recolor{dna.cell[0]}{state=done}
\recolor{dna.cell[1]}{state=current}
\narrate{Vị trí 1: T $\neq$ A. Bắt đầu đoạn lặp mới. current = 1, max = 1.}

\step
\recolor{dna.cell[1]}{state=done}
\recolor{dna.cell[2]}{state=current}
\apply{vars.cell[0]}{value=2}
\apply{vars.cell[1]}{value=2}
\narrate{Vị trí 2: T $=$ T. Đoạn lặp kéo dài thành 2. max = 2.}

\step
\recolor{dna.cell[2]}{state=done}
\recolor{dna.cell[3]}{state=current}
\apply{vars.cell[0]}{value=1}
\narrate{Vị trí 3: C $\neq$ T. Bắt đầu đoạn lặp mới. current = 1, max = 2.}

\step
\recolor{dna.cell[3]}{state=done}
\recolor{dna.cell[4]}{state=current}
\narrate{Vị trí 4: G $\neq$ C. Bắt đầu đoạn lặp mới. current = 1, max = 2.}

\step
\recolor{dna.cell[4]}{state=done}
\recolor{dna.cell[5]}{state=current}
\apply{vars.cell[0]}{value=2}
\narrate{Vị trí 5: G $=$ G. Đoạn lặp kéo dài thành 2. max = 2.}

\step
\recolor{dna.cell[5]}{state=done}
\recolor{dna.cell[6]}{state=current}
\apply{vars.cell[0]}{value=3}
\apply{vars.cell[1]}{value=3}
\recolor{vars.cell[1]}{state=good}
\narrate{Vị trí 6: G $=$ G. Đoạn lặp kéo dài thành 3! Max mới = 3.}

\step
\recolor{dna.cell[6]}{state=done}
\recolor{dna.cell[7]}{state=current}
\apply{vars.cell[0]}{value=1}
\narrate{Vị trí 7: A $\neq$ G. Bắt đầu đoạn lặp mới. current = 1, max = 3.}

\step
\recolor{dna.cell[7]}{state=done}
\recolor{dna.range[4:6]}{state=good}
\narrate{Duyệt xong. Đoạn lặp dài nhất: 3 (GGG tại vị trí 4--6). Đáp án: $\mathbf{3}$.}
\end{animation}

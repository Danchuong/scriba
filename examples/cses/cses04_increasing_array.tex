\begin{animation}[id="increasing-array", label="Minimum Moves to Make Non-Decreasing"]

\compute{
arr = [3, 2, 5, 1, 7]
n = len(arr)
}

\shape{a}{Array}{size=5, data=[3, 2, 5, 1, 7], labels="0..4", label="Array"}
\shape{cost}{Array}{size=1, data=[0], label="Total moves"}

\step
\narrate{Biến mảng thành không giảm với số lần tăng ít nhất. Mảng: $[3, 2, 5, 1, 7]$.}

\step
\recolor{a.cell[0]}{state=done}
\narrate{Vị trí $0$: giá trị $3$. Phần tử đầu tiên, luôn hợp lệ. Max hiện tại $= 3$.}

\step
\recolor{a.cell[1]}{state=error}
\narrate{Vị trí $1$: giá trị $2$ $<$ giá trị trước $3$. Vi phạm! Cần tăng thêm $3 - 2 = 1$.}

\step
\apply{a.cell[1]}{value=3}
\apply{cost.cell[0]}{value=1}
\recolor{a.cell[1]}{state=done}
\narrate{Gán $a[1] = 3$. Tổng thao tác $= 1$. Max hiện tại $= 3$.}

\step
\recolor{a.cell[2]}{state=current}
\narrate{Vị trí $2$: giá trị $5$ $\geq$ $3$. Không cần tăng. Max hiện tại $= 5$.}

\step
\recolor{a.cell[2]}{state=done}
\recolor{a.cell[3]}{state=error}
\narrate{Vị trí $3$: giá trị $1$ $<$ giá trị trước $5$. Vi phạm! Cần tăng thêm $5 - 1 = 4$.}

\step
\apply{a.cell[3]}{value=5}
\apply{cost.cell[0]}{value=5}
\recolor{a.cell[3]}{state=done}
\narrate{Gán $a[3] = 5$. Tổng thao tác $= 1 + 4 = 5$. Max hiện tại $= 5$.}

\step
\recolor{a.cell[4]}{state=current}
\narrate{Vị trí $4$: giá trị $7$ $\geq$ $5$. Không cần tăng.}

\step
\recolor{a.cell[4]}{state=done}
\recolor{a.all}{state=good}
\recolor{cost.cell[0]}{state=good}
\narrate{Mảng giờ đã không giảm: $[3, 3, 5, 5, 7]$. Tổng thao tác $= \mathbf{5}$.}
\end{animation}

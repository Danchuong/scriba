\begin{animation}[id="permutations", label="Beautiful Permutation Construction"]

\compute{
  n = 8
  evens = [i for i in range(2, n + 1, 2)]
  odds = [i for i in range(1, n + 1, 2)]
  result = evens + odds
}

\shape{perm}{Array}{size=8, data=[0, 0, 0, 0, 0, 0, 0, 0], labels="0..7", label="Permutation"}
\shape{diffs}{Array}{size=7, data=[0, 0, 0, 0, 0, 0, 0], labels="0..6", label="Adjacent diffs"}

\step
\narrate{Xây dựng hoán vị đẹp của $[1..8]$: không có hai phần tử liền kề nào có hiệu bằng $1$. Chiến lược: đặt tất cả số chẵn trước, rồi đến số lẻ.}

\step
\apply{perm.cell[0]}{value=2}
\apply{perm.cell[1]}{value=4}
\apply{perm.cell[2]}{value=6}
\apply{perm.cell[3]}{value=8}
\recolor{perm.range[0:3]}{state=current}
\narrate{Đặt các số chẵn: $[2, 4, 6, 8]$. Mỗi cặp liền kề có hiệu bằng $2$. An toàn!}

\step
\apply{diffs.cell[0]}{value=2}
\apply{diffs.cell[1]}{value=2}
\apply{diffs.cell[2]}{value=2}
\recolor{diffs.range[0:2]}{state=done}
\recolor{perm.range[0:3]}{state=done}
\narrate{Hiệu trong dãy chẵn: $|4-2| = 2$, $|6-4| = 2$, $|8-6| = 2$. Tất cả $\geq 2$. Tốt.}

\step
\apply{perm.cell[4]}{value=1}
\apply{perm.cell[5]}{value=3}
\apply{perm.cell[6]}{value=5}
\apply{perm.cell[7]}{value=7}
\recolor{perm.range[4:7]}{state=current}
\narrate{Đặt các số lẻ: $[1, 3, 5, 7]$. Mỗi cặp liền kề có hiệu bằng $2$. An toàn!}

\step
\apply{diffs.cell[3]}{value=7}
\apply{diffs.cell[4]}{value=2}
\apply{diffs.cell[5]}{value=2}
\apply{diffs.cell[6]}{value=2}
\recolor{diffs.cell[3]}{state=good}
\recolor{diffs.range[4:6]}{state=done}
\highlight{perm.cell[3]}
\highlight{perm.cell[4]}
\narrate{Ranh giới: $|8 - 1| = 7$. Hiệu trong dãy lẻ: đều bằng $2$. Ranh giới quan trọng nhất vẫn an toàn vì $7 \neq 1$.}

\step
\recolor{perm.all}{state=good}
\recolor{diffs.all}{state=good}
\narrate{Hoán vị đẹp: $[2, 4, 6, 8, 1, 3, 5, 7]$. Tất cả hiệu liền kề $\geq 2$: $[2, 2, 2, 7, 2, 2, 2]$. Hoàn thành!}
\end{animation}

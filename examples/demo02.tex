\begin{animation}[id="bubblesort", label="Bubble Sort First Pass"]
\shape{a}{Array}{size=5, data=[5,3,8,1,4], labels="0..4"}

\step
\recolor{a.cell[0]}{state=current}
\recolor{a.cell[1]}{state=current}
\narrate{Compare a[0]=5 and a[1]=3. Since 5>3, swap them.}

\step
\apply{a.cell[0]}{value=3}
\apply{a.cell[1]}{value=5}
\recolor{a.cell[0]}{state=done}
\recolor{a.cell[1]}{state=current}
\recolor{a.cell[2]}{state=current}
\narrate{After swap: [3,5,...]. Compare a[1]=5 and a[2]=8. No swap needed.}

\step
\recolor{a.cell[1]}{state=done}
\recolor{a.cell[2]}{state=current}
\recolor{a.cell[3]}{state=current}
\narrate{Compare a[2]=8 and a[3]=1. Since 8>1, swap them.}

\step
\apply{a.cell[2]}{value=1}
\apply{a.cell[3]}{value=8}
\recolor{a.cell[2]}{state=done}
\recolor{a.cell[3]}{state=current}
\recolor{a.cell[4]}{state=current}
\narrate{After swap: [...,1,8,4]. Compare a[3]=8 and a[4]=4. Since 8>4, swap. End of pass: [3,5,1,4,8].}
\end{animation}

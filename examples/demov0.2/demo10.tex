\begin{animation}[id="dijkstra", label="Dijkstra Shortest Path"]
\shape{G}{Graph}{nodes=[S,A,B,C,D,T], edges=[(S,A),(S,B),(A,C),(B,C),(B,D),(C,T),(D,T)], directed=true}

\step
\recolor{G.node[S]}{state=current}
\narrate{Initialize source S with distance 0. All others are infinity.}

\step
\recolor{G.node[S]}{state=done}
\recolor{G.edge[(S,A)]}{state=current}
\recolor{G.edge[(S,B)]}{state=current}
\recolor{G.node[A]}{state=current}
\recolor{G.node[B]}{state=current}
\narrate{Relax edges from S: dist(A)=4, dist(B)=2. Pick B (smallest).}

\step
\recolor{G.node[B]}{state=done}
\recolor{G.edge[(S,B)]}{state=done}
\recolor{G.edge[(B,C)]}{state=current}
\recolor{G.edge[(B,D)]}{state=current}
\recolor{G.node[C]}{state=current}
\recolor{G.node[D]}{state=current}
\narrate{Relax from B: dist(C)=5, dist(D)=7. Pick A next (dist=4).}

\step
\recolor{G.node[A]}{state=done}
\recolor{G.edge[(S,A)]}{state=done}
\recolor{G.edge[(A,C)]}{state=done}
\recolor{G.node[C]}{state=done}
\recolor{G.edge[(B,C)]}{state=dim}
\narrate{Relax from A: dist(C)=min(5,4+1)=5 unchanged. Pick C next.}

\step
\recolor{G.edge[(C,T)]}{state=done}
\recolor{G.node[D]}{state=done}
\recolor{G.edge[(B,D)]}{state=dim}
\recolor{G.edge[(D,T)]}{state=dim}
\recolor{G.node[T]}{state=good}
\narrate{Relax from C: dist(T)=8. From D: dist(T)=min(8,9)=8. Shortest path S-B-C-T = 8.}
\end{animation}

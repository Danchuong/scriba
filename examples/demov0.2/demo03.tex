\begin{animation}[id="fibdp", label="Fibonacci DP"]
\shape{dp}{DPTable}{n=6, labels="dp[0]..dp[5]"}

\step
\recolor{dp.cell[0]}{state=current}
\apply{dp.cell[0]}{value=0}
\narrate{Base case: dp[0] = 0.}

\step
\recolor{dp.cell[0]}{state=done}
\recolor{dp.cell[1]}{state=current}
\apply{dp.cell[1]}{value=1}
\narrate{Base case: dp[1] = 1.}

\step
\recolor{dp.cell[1]}{state=done}
\recolor{dp.cell[2]}{state=current}
\apply{dp.cell[2]}{value=1}
\narrate{dp[2] = dp[0] + dp[1] = 0 + 1 = 1.}

\step
\recolor{dp.cell[2]}{state=done}
\recolor{dp.cell[3]}{state=current}
\apply{dp.cell[3]}{value=2}
\narrate{dp[3] = dp[1] + dp[2] = 1 + 1 = 2.}

\step
\recolor{dp.cell[3]}{state=done}
\recolor{dp.cell[4]}{state=current}
\apply{dp.cell[4]}{value=3}
\recolor{dp.cell[5]}{state=current}
\apply{dp.cell[5]}{value=5}
\narrate{dp[4] = dp[2] + dp[3] = 1 + 2 = 3. dp[5] = dp[3] + dp[4] = 2 + 3 = 5.}
\end{animation}

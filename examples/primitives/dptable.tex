\begin{animation}[id="dptable-demo", label="DPTable -- Minimum cost path"]
\shape{cost}{Array}{size=6, data=[0,3,2,6,1,4], labels="0..5", label="$cost$"}
\shape{dp}{DPTable}{n=6, data=["","","","","",""], labels="0..5", label="$dp$"}

\step
\narrate{Given an array $cost$ of length $N=6$. Find the minimum total cost to reach index $N-1$, starting from index $0$. At each step you can jump $1$ or $2$ positions forward, paying $cost[j]$ when you land on index $j$.}

\step
\recolor{cost.cell[0]}{state=current}
\recolor{dp.cell[0]}{state=done}
\apply{dp.cell[0]}{value=0}
\narrate{Base case: $dp[0] = 0$. We start at index $0$ with zero cost (we are already here).}

\step
\cursor{cost.cell}{1}
\recolor{dp.cell[1]}{state=current}
\annotate{dp.cell[1]}{label="+3", arrow_from="dp.cell[0]", color=good}
\narrate{$dp[1]$: only reachable from index $0$. Cost $= dp[0] + cost[1] = 0 + 3 = 3$.}

\step
\apply{dp.cell[1]}{value=3}
\recolor{dp.cell[1]}{state=done}
\cursor{cost.cell}{2}
\recolor{dp.cell[2]}{state=current}
\annotate{dp.cell[2]}{label="+2", arrow_from="dp.cell[0]", color=info}
\annotate{dp.cell[2]}{label="+2", arrow_from="dp.cell[1]", color=good}
\narrate{$dp[2]$: from index $0$: $0 + 2 = 2$; from index $1$: $3 + 2 = 5$. Best: $\min(2, 5) = 2$.}

\step
\apply{dp.cell[2]}{value=2}
\recolor{dp.cell[2]}{state=done}
\cursor{cost.cell}{3}
\recolor{dp.cell[3]}{state=current}
\annotate{dp.cell[3]}{label="+6", arrow_from="dp.cell[1]", color=info}
\annotate{dp.cell[3]}{label="+6", arrow_from="dp.cell[2]", color=good}
\narrate{$dp[3]$: from index $1$: $3 + 6 = 9$; from index $2$: $2 + 6 = 8$. Best: $\min(9, 8) = 8$.}

\step
\apply{dp.cell[3]}{value=8}
\recolor{dp.cell[3]}{state=done}
\cursor{cost.cell}{4}
\recolor{dp.cell[4]}{state=current}
\annotate{dp.cell[4]}{label="+1", arrow_from="dp.cell[2]", color=info}
\annotate{dp.cell[4]}{label="+1", arrow_from="dp.cell[3]", color=good}
\narrate{$dp[4]$: from index $2$: $2 + 1 = 3$; from index $3$: $8 + 1 = 9$. Best: $\min(3, 9) = 3$.}

\step
\apply{dp.cell[4]}{value=3}
\recolor{dp.cell[4]}{state=done}
\cursor{cost.cell}{5}
\recolor{dp.cell[5]}{state=current}
\annotate{dp.cell[5]}{label="+4", arrow_from="dp.cell[3]", color=info}
\annotate{dp.cell[5]}{label="+4", arrow_from="dp.cell[4]", color=good}
\narrate{$dp[5]$: from index $3$: $8 + 4 = 12$; from index $4$: $3 + 4 = 7$. Best: $\min(12, 7) = 7$.}

\step
\apply{dp.cell[5]}{value=7}
\recolor{dp.cell[5]}{state=done}
\recolor{cost.cell[5]}{state=idle}
\recolor{dp.cell[0]}{state=path}
\recolor{dp.cell[2]}{state=path}
\recolor{dp.cell[4]}{state=path}
\recolor{dp.cell[5]}{state=path}
\recolor{cost.cell[0]}{state=path}
\recolor{cost.cell[2]}{state=path}
\recolor{cost.cell[4]}{state=path}
\recolor{cost.cell[5]}{state=path}
\reannotate{dp.cell[2]}{color=path, arrow_from="dp.cell[0]"}
\reannotate{dp.cell[4]}{color=path, arrow_from="dp.cell[2]"}
\reannotate{dp.cell[5]}{color=path, arrow_from="dp.cell[4]"}
\narrate{$dp[5] = 7$. Optimal path: $0 \to 2 \to 4 \to 5$ with total cost $0 + 2 + 1 + 4 = 7$.}
\end{animation}
